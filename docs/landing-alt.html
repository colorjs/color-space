One conversion graph. Four ways to run it.

Use scalar JavaScript for individual colors, WASM for buffers, generated shader chunks for the GPU, or a .cube LUT for video hosts.

import space from 'color-space'

space.rgb.oklch(255, 128, 0)
// → [0.732, 0.186, 53]

space.slog3.rec709(r, g, b)
import space, { alloc } from 'color-space/wasm'

const pixels = alloc(n)
// write RGB pixels into the shared buffer
space.rgb.oklch(pixels) // in place, zero-copy
import { glsl, wgsl } from 'color-space/gl'
import oklch from 'color-space/gl/oklch'

glsl(oklch, 'rgb')
wgsl(oklch, 'rgb')
import { cube } from 'color-space/lut'

cube(space.slog3, space.rec709)
// verified .cube — Resolve, Premiere, FCP, OBS, ffmpeg
